IF DB_ID('PlataformaCursos') IS NULL
BEGIN
    CREATE DATABASE PlataformaCursos;
END
GO

USE PlataformaCursos;
GO

IF OBJECT_ID('dbo.Usuarios', 'U') IS NULL
BEGIN
    CREATE TABLE dbo.Usuarios (
        IdUsuario INT IDENTITY(1,1) PRIMARY KEY,
        Nombre NVARCHAR(100) NOT NULL,
        Apellidos NVARCHAR(150) NOT NULL,
        Email NVARCHAR(150) NOT NULL UNIQUE,
        PasswordHash NVARCHAR(255) NOT NULL,
        Rol NVARCHAR(20) NOT NULL CHECK (Rol IN ('ALUMNO', 'PROFESOR', 'ADMIN')),
        FechaRegistro DATETIME NOT NULL DEFAULT GETDATE()
    );
END
GO

IF OBJECT_ID('dbo.Cursos', 'U') IS NULL
BEGIN
    CREATE TABLE dbo.Cursos (
        IdCurso INT IDENTITY(1,1) PRIMARY KEY,
        Titulo NVARCHAR(200) NOT NULL,
        Descripcion NVARCHAR(MAX) NULL,
        IdProfesor INT NOT NULL,
        FechaCreacion DATETIME NOT NULL DEFAULT GETDATE(),
        Activo BIT NOT NULL DEFAULT 1,
        Precio DECIMAL(10,2) NOT NULL DEFAULT 0,
        AceptaEfectivo BIT NOT NULL DEFAULT 1,
        AceptaTarjeta BIT NOT NULL DEFAULT 1,
        AceptaTransferencia BIT NOT NULL DEFAULT 1,
        CONSTRAINT FK_Curso_Profesor FOREIGN KEY (IdProfesor) REFERENCES dbo.Usuarios(IdUsuario) ON DELETE NO ACTION
    );
END
GO

IF OBJECT_ID('dbo.Lecciones', 'U') IS NULL
BEGIN
    CREATE TABLE dbo.Lecciones (
        IdLeccion INT IDENTITY(1,1) PRIMARY KEY,
        IdCurso INT NOT NULL,
        Titulo NVARCHAR(200) NOT NULL,
        Contenido NVARCHAR(MAX) NULL,
        PreguntaEvaluacion NVARCHAR(500) NOT NULL DEFAULT 'żQué has aprendido en esta lección?',
        RespuestaCorrecta NVARCHAR(250) NOT NULL DEFAULT 'OK',
        Orden INT NOT NULL,
        FechaCreacion DATETIME NOT NULL DEFAULT GETDATE(),
        CONSTRAINT FK_Leccion_Curso FOREIGN KEY (IdCurso) REFERENCES dbo.Cursos(IdCurso) ON DELETE CASCADE
    );
END
GO

IF OBJECT_ID('dbo.Inscripciones', 'U') IS NULL
BEGIN
    CREATE TABLE dbo.Inscripciones (
        IdInscripcion INT IDENTITY(1,1) PRIMARY KEY,
        IdUsuario INT NOT NULL,
        IdCurso INT NOT NULL,
        FechaInscripcion DATETIME NOT NULL DEFAULT GETDATE(),
        Estado NVARCHAR(20) NOT NULL DEFAULT 'ACTIVO' CHECK (Estado IN ('ACTIVO', 'FINALIZADO', 'CANCELADO')),
        MetodoPago NVARCHAR(20) NOT NULL DEFAULT 'SIN_COSTE',
        ImportePagado DECIMAL(10,2) NOT NULL DEFAULT 0,
        CONSTRAINT FK_Inscripcion_Usuario FOREIGN KEY (IdUsuario) REFERENCES dbo.Usuarios(IdUsuario) ON DELETE CASCADE,
        CONSTRAINT FK_Inscripcion_Curso FOREIGN KEY (IdCurso) REFERENCES dbo.Cursos(IdCurso) ON DELETE CASCADE,
        CONSTRAINT UQ_Usuario_Curso UNIQUE (IdUsuario, IdCurso)
    );
END
GO

IF OBJECT_ID('dbo.Progreso', 'U') IS NULL
BEGIN
    CREATE TABLE dbo.Progreso (
        IdProgreso INT IDENTITY(1,1) PRIMARY KEY,
        IdUsuario INT NOT NULL,
        IdLeccion INT NOT NULL,
        Completado BIT NOT NULL DEFAULT 0,
        FechaUltimaActualizacion DATETIME NOT NULL DEFAULT GETDATE(),
        CONSTRAINT FK_Progreso_Usuario FOREIGN KEY (IdUsuario) REFERENCES dbo.Usuarios(IdUsuario) ON DELETE CASCADE,
        CONSTRAINT FK_Progreso_Leccion FOREIGN KEY (IdLeccion) REFERENCES dbo.Lecciones(IdLeccion) ON DELETE CASCADE,
        CONSTRAINT UQ_Usuario_Leccion UNIQUE (IdUsuario, IdLeccion)
    );
END
GO
